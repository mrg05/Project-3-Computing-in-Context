#Step 5: Calculating Mean, Median, and Mode with Standard Python Also, round up the 3 values upto 2 decimal points

#Definitions of these terms: Mean: Average performance score across students. Median: Middle score, useful for skewed data. Mode: Most frequently occurring score, highlighting common performance trends.

#Convert the 'Sleep_Hours' column to a list
Hours = data['Sleep_Hours'].tolist()

#Calculate Mean
mean_python = sum(Hours) / len(Hours)
mean_python = round(mean_python, 2)

#Calculate Median
sorted_Hours = sorted(Hours)
n = len(sorted_Hours)
if n % 2 == 1:
    median_python = sorted_Hours[n // 2]
else:
    median_python = (sorted_Hours[n // 2 - 1] + sorted_Hours[n // 2]) / 2
median_python = round(median_python, 2)

#Calculate Mode
frequency = {}
for Hours in Hours:
    frequency[Hours] = frequency.get(Hours, 0) + 1
mode_python = max(frequency, key=frequency.get)

print("Mean (Standard Python):", mean_python)
print("Median (Standard Python):", median_python)
print("Mode (Standard Python):", mode_python)

#Step 6: Data Visualization (Gender vs. Exam Scores) A simple text-based bar chart will be created to represent the distribution of values in the Sleep_Hours column. This visualization uses only the Python standard library, as specified in the project guidelines. The chart is dynamically generated, ensuring it adjusts automatically if the dataset is modified

# Group data by gender and count the number of exam scores
gender_counts = {}
for _, row in data.iterrows():
    gender = row['Gender']
    score = row['Exam_Score']
    if not pd.isnull(gender) and not pd.isnull(score):  # Skip rows with missing Gender or Exam_Score
        gender_counts[gender] = gender_counts.get(gender, 0) + 1

# Find the maximum count for normalization
max_count = max(gender_counts.values())

# Generate a text-based bar chart
print("Gender vs Exam Score (Count):")
print("-" * 40)
for gender, count in gender_counts.items():
    bar_length = int((count / max_count) * 30)  
    print(f"{gender:<10} | {'*' * bar_length} ({count})")
